\tikzstyle{btreeptr} = [minimum height=1.5em]
\tikzstyle{btreeval} = [minimum size=1.5em]

\newcommand{\xyshift}[3]{%
  \begin{scope}[xshift=#1, yshift=#2]
    #3
  \end{scope}%
}

\newcommand{\btreenodethree}[4]{%
  \matrix[draw, med grayed, semithick, ampersand replacement=\&, inner sep=0pt, outer sep=0pt, anchor=center] (#1)
  {
    \node[btreeptr] (#1-1) {}; \& \node[btreeval] (#1-a) {#2}; \&
    \node[btreeptr] (#1-2) {}; \& \node[btreeval] (#1-b) {#3}; \&
    \node[btreeptr] (#1-3) {}; \& \node[btreeval] (#1-c) {#4}; \&
    \node[btreeptr] (#1-4) {}; \\
  };
}

\newcommand{\btreenodetwo}[3]{%
  \matrix[draw, med grayed, semithick, ampersand replacement=\&, inner sep=0pt, outer sep=0pt, anchor=center] (#1)
  {
    \node[btreeptr] (#1-1) {}; \& \node[btreeval] (#1-a) {#2}; \&
    \node[btreeptr] (#1-2) {}; \& \node[btreeval] (#1-b) {#3}; \&
    \node[btreeptr] (#1-3) {}; \\
  };
}

\newcommand{\btreenodeone}[2]{%
  \matrix[draw, med grayed, semithick, ampersand replacement=\&, inner sep=0pt, outer sep=0pt, anchor=center] (#1)
  {
    \node[btreeptr] (#1-1) {}; \& \node[btreeval] (#1-a) {#2}; \&
    \node[btreeptr] (#1-2) {}; \\
  };
}

\begin{tikzpicture}[arrow/.append style = {shorten <=-3pt}]

\node[outer] (pic a) {
\begin{tikzpicture}
	\btreenodetwo{root}{2}{4}
    \xyshift{-10mm}{-15mm}{\btreenodeone{n1}{1}}
    \xyshift{-0mm}{-15mm}{\btreenodeone{n2}{13}}
    \xyshift{10mm}{-15mm}{\btreenodeone{n3}{15}}
	\foreach \x in {1,2,3} {
		\draw[arrow] (root-\x.south) -- (n\x.north);
	}
\end{tikzpicture}
};

\node[outer, right=of pic a] (pic b) {
\begin{tikzpicture}
	\btreenodeone{root}{2}
    \xyshift{-7mm}{-15mm}{\btreenodeone{n1}{1}}
    \xyshift{7mm}{-15mm}{\btreenodethree{n2}{3}{4}{5}}
	\foreach \x in {1,2} {
		\draw[arrow] (root-\x.south) -- (n\x.north);
	}
\end{tikzpicture}
};

\node[outer, right=of pic b] (pic c) {
\begin{tikzpicture}
	\btreenodeone{root}{3}
    \xyshift{-8mm}{-15mm}{\btreenodetwo{n1}{1}{2}}
    \xyshift{8mm}{-15mm}{\btreenodetwo{n2}{4}{5}}
	\foreach \x in {1,2} {
		\draw[arrow] (root-\x.south) -- (n\x.north);
	}
\end{tikzpicture}
};

\node[outer, right=of pic c] (pic d) {
\begin{tikzpicture}
	\btreenodeone{root}{4}

	}
\end{tikzpicture}

\foreach \x in {a,b,c,d} {
